# Meta-tags

```html

        <title>Page title</title>
        <link rel="shortcut icon" href="/icon.ico" />
        
        <!-- Favicons -->
        <link rel="icon" href="/icon.png" />
        
        <!-- Apple Touch Icons -->
        <link rel="apple-touch-icon" href="/icon.png" />
        
        <!-- Primary Meta Tags -->
        <meta name="title" content="Page title" />
        <meta name="description" content="Page description" />
        <meta name="viewport" content="width=device-width, initial-scale=1"/>
        
        <!-- Open Graph / Facebook -->
        <meta property="og:type" content="article" />
        <meta property="article:publisher" content="Rafael Araujo" />
        <meta property="article:published_time" content="2022-08-05T21:32Z" />
        <meta property="article:modified_time" content="2022-08-05T25:21Z" />
        
        <meta property="og:site_name" content="Site title" />
        <meta property="og:url" content="https://github.com/Rafael-Araujo-dev/Meta-tags/edit/main/README.md" />
        <meta property="og:locale" content="pt_BR" />
        <meta property="og:title" content="Page title" />
        <meta property="og:description" content="Page description" />
        
        <meta property="og:image" content="/image.jpeg" />
        <meta property="og:image:secure_url" content="/image.jpeg" />
        <meta property="og:image:type" content="image/jpeg" />
        <meta property="og:image:type" content="image/png" />
        <meta property="og:image:width" content="1280" />
        <meta property="og:image:height" content="1720" />
        <meta property="og:image:alt" content="Image Alt text" />
        
        <!-- Twitter -->
        <meta property="twitter:card" content="summary_large_image" />
        <meta property="twitter:url" content="https://github.com/Rafael-Araujo-dev/Meta-tags/edit/main/README.md" />
        <meta property="twitter:title" content="Page title" />
        <meta property="twitter:description" content="Page description" />
        <meta property="twitter:image" content="/image.jpeg" />

```
